   123→// ─── 3D Tilt Card Hook (with touch support) ───
   124→export function useTilt() {
   125→  const [style, setStyle] = useState<React.CSSProperties>({})
   126→  const onMove = useCallback((e: React.MouseEvent<HTMLElement>) => {
   127→    const el = e.currentTarget
   128→    const rect = el.getBoundingClientRect()
   129→    const x = (e.clientX - rect.left) / rect.width - 0.5
   130→    const y = (e.clientY - rect.top) / rect.height - 0.5
   131→    setStyle({ transform: `perspective(800px) rotateY(${x * 10}deg) rotateX(${-y * 10}deg) translateZ(12px) scale(1.03)` })
   132→  }, [])
   133→  const onTouch = useCallback((e: React.TouchEvent<HTMLElement>) => {
   134→    const el = e.currentTarget
   135→    const rect = el.getBoundingClientRect()
   136→    const t = e.touches[0]
   137→    const x = (t.clientX - rect.left) / rect.width - 0.5
   138→    const y = (t.clientY - rect.top) / rect.height - 0.5
   139→    setStyle({ transform: `perspective(800px) rotateY(${x * 6}deg) rotateX(${-y * 6}deg) translateZ(8px) scale(1.02)` })
   140→  }, [])
   141→  const onLeave = useCallback(() => setStyle({ transform: 'perspective(800px) rotateY(0deg) rotateX(0deg) translateZ(0px) scale(1)', transition: 'transform 0.4s cubic-bezier(0.16, 1, 0.3, 1)' }), [])
   142→  return { style, onMove, onTouch, onLeave }
   143→}

   145→// ─── Live Clock ───
   146→export function LiveClock() {
   147→  const [t, setT] = useState<Date | null>(null)
   148→  useEffect(() => {
   149→    setT(new Date())
   150→    const i = setInterval(() => setT(new Date()), 1000)
   151→    return () => clearInterval(i)
   152→  }, [])
   153→  return (
   154→    <span className="text-[11px] text-slate-400 tabular-nums font-medium">
   155→      {t ? t.toLocaleTimeString('en-IN', { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: true }) : '--:--:-- --'}
   156→    </span>
   157→  )
   158→}
